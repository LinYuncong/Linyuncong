--- Table: users ---
CREATE TABLE users (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,

    -- 核心身份
    username        TEXT UNIQUE NOT NULL,       -- 用户名（登录用）
    display_name    TEXT DEFAULT '',            -- 显示名 frontend: display_name
    acct            TEXT NOT NULL,              -- 完整账号, 单实例时 = username frontend: acct
    note            TEXT DEFAULT '',            -- 个人简介 (HTML) frontend: note
    locked          INTEGER DEFAULT 0,          -- 是否锁定（需要关注审核）frontend: locked
    bot             INTEGER DEFAULT 0,          -- 是否为机器人 frontend: bot

    header          TEXT DEFAULT '',            -- 背景图 URL frontend: header
    url             TEXT DEFAULT '',            -- 个人主页 URL frontend: url

    -- 统计 (可实时计算)
    followers_count   INTEGER DEFAULT 0,        -- frontend: followers_count
    following_count   INTEGER DEFAULT 0,        -- frontend: following_count
    statuses_count    INTEGER DEFAULT 0,        -- frontend: statuses_count

    -- 权限
    role            TEXT DEFAULT 'user',        -- user / admin / moderator frontend: role.permissions

    -- 自定义字段 (JSON 数组) frontend: fields

    fields          TEXT DEFAULT '[]',

    -- 时间
    last_status_at  TEXT,                       -- frontend: last_status_at
    created_at      TEXT DEFAULT (datetime('now')), -- frontend: created_at
    updated_at      TEXT DEFAULT (datetime('now')),

    -- 认证
    email           TEXT UNIQUE NOT NULL,       -- 登录邮箱
    password_hash   TEXT NOT NULL,              -- bcrypt 加密密码
